75 int timer_expired( timer_id timer ) {
76  assert( timer != NULL );
77 
78  double current_time = get_current_time();
79  double time_diff = current_time - timer->reset_time;
80  int expired = time_diff * MILLISECONDS >= timer->milliseconds;

82  if ( expired ) {

126 int clock_gettime( int X, struct timeval *tv ) {
127  LARGE_INTEGER t;
128  FILETIME f;
129  double microseconds;
130  static LARGE_INTEGER offset;
131  static double frequencyToMicroseconds;
132  static int initialized = 0;
133  static BOOL usePerformanceCounter = 0;
134 
135  if ( !initialized ) {
136  LARGE_INTEGER performanceFrequency;
137  initialized = 1;
138  usePerformanceCounter = QueryPerformanceFrequency( &performanceFrequency );
139  if ( usePerformanceCounter ) {
140  QueryPerformanceCounter( &offset );
141  frequencyToMicroseconds = (double) performanceFrequency.QuadPart / 1000000.;
142  }
143  else {
144  offset = getFILETIMEoffset();
145  frequencyToMicroseconds = 10.;
146  }
147  }
148  if ( usePerformanceCounter ) QueryPerformanceCounter( &t );
149  else {
150  GetSystemTimeAsFileTime( &f );
151  t.QuadPart = f.dwHighDateTime;
152  t.QuadPart <<= 32;
153  t.QuadPart |= f.dwLowDateTime;
154  }
155 
156  t.QuadPart -= offset.QuadPart;
157  microseconds = (double) t.QuadPart / frequencyToMicroseconds;
158  t.QuadPart = (LONGLONG) microseconds;
159  tv->tv_sec = (long) ( t.QuadPart / 1000000 );
160  tv->tv_usec = t.QuadPart % 1000000;
161  return ( 0 );
162 }
